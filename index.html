// =============================== // SNOCROSS GAME ENGINE (C3) // =============================== // Global game state const Game = { series: "12-14", maxRPM: 6450, halfGas: 1, exhaust: "stock", hasCAPRO: 0, hasStuds: 0, throttle: 0, rpm: 0, speed: 0, angle: 0, onGround: 0, keyGas: 0, keyBrake: 0, keyLeft: 0, keyRight: 0 }; // =============================== // INPUT SYSTEM // =============================== document.addEventListener("keydown", e => { if (e.key === "ArrowUp") Game.keyGas = 1; if (e.key === "ArrowDown") Game.keyBrake = 1; if (e.key === "ArrowLeft") Game.keyLeft = 1; if (e.key === "ArrowRight") Game.keyRight = 1; }); document.addEventListener("keyup", e => { if (e.key === "ArrowUp") Game.keyGas = 0; if (e.key === "ArrowDown") Game.keyBrake = 0; if (e.key === "ArrowLeft") Game.keyLeft = 0; if (e.key === "ArrowRight") Game.keyRight = 0; }); // =============================== // MAIN UPDATE LOOP // =============================== runOnStartup(async runtime => { runtime.addEventListener("tick", () => { const dt = runtime.dt; // --------------------------- // GAS / BROMs // --------------------------- if (Game.keyGas) Game.throttle += 1.5 * dt; else Game.throttle -= 2 * dt; if (Game.keyBrake) Game.throttle -= 3 * dt; if (Game.halfGas) Game.throttle = Math.min(Math.max(Game.throttle, 0), 0.5); else Game.throttle = Math.min(Math.max(Game.throttle, 0), 1); // --------------------------- // RPM // --------------------------- Game.rpm = Game.throttle * Game.maxRPM; // --------------------------- // ACCELERATION // --------------------------- let accel = 40; if (Game.series === "PRO_LITE" || Game.series === "LITE") accel *= 1.2; if (Game.series === "OPEN") accel *= 1.3; if (Game.exhaust === "ggb") accel *= 1.05; if (Game.exhaust === "custom") accel *= 1.1; if (Game.hasStuds) accel *= 1.03; if (Game.hasCAPRO) accel *= 1.02; Game.speed += Game.throttle * accel * dt; Game.speed -= Game.speed * 0.1 * dt; // --------------------------- // APPLY PHYSICS TO PLAYER // --------------------------- const sled = runtime.objects.PlayerSled.getFirstInstance(); if (sled) sled.applyForce(Game.speed * 10, 0); // --------------------------- // WHIP // --------------------------- if (!Game.onGround) { if (Game.keyLeft) Game.angle -= 120 * dt; if (Game.keyRight) Game.angle += 120 * dt; } else { Game.angle *= 0.9; } Game.angle = Math.max(-45, Math.min(45, Game.angle)); if (sled) sled.angle = Game.angle; // --------------------------- // CAMERA // --------------------------- const cam = runtime.objects.CameraDummy.getFirstInstance(); if (cam && sled) { cam.x = sled.x + 300; cam.y = sled.y - 150; } }); });
