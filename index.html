using UnityEngine; using System.Collections.Generic; public class DragCarGame : MonoBehaviour { public Car playerCar; public Car enemyCar; public Transform finishLine; public float finishLineRadius = 5f; // Made configurable public float raceResetDelay = 3f; private bool raceStarted = false; private bool raceFinished = false; void Start() { ValidateReferences(); } void ValidateReferences() { if (playerCar == null || enemyCar == null || finishLine == null) { Debug.LogError("DragCarGame: Missing required references!"); } } void Update() { if (!raceStarted && !raceFinished && Input.GetKeyDown(KeyCode.Space)) { StartRace(); } if (raceStarted) { CheckWinner(); } } void StartRace() { raceStarted = true; raceFinished = false; playerCar.ResetCar(); enemyCar.ResetCar(); enemyCar.EnableAI(true); // Start AI Debug.Log("Race Started!"); } void CheckWinner() { if (raceFinished) return; float playerDistance = Vector3.Distance(playerCar.transform.position, finishLine.position); float enemyDistance = Vector3.Distance(enemyCar.transform.position, finishLine.position); if (playerDistance < finishLineRadius) { DecideWinner("PLAYER"); return; } if (enemyDistance < finishLineRadius) { DecideWinner("ENEMY"); } } void DecideWinner(string winner) { raceStarted = false; raceFinished = true; enemyCar.EnableAI(false); Debug.Log(winner + " WINS!"); Invoke(nameof(ResetRace), raceResetDelay); } void ResetRace() { raceFinished = false; Debug.Log("Press SPACE to start a new race"); } } [System.Serializable] public class CarStats { public float horsepower = 200f; public float weight = 1200f; public float traction = 1f; public float aerodynamics = 1f; public float GetAcceleration() { return (horsepower / weight) * traction * 50f; } public float GetTopSpeed() { return horsepower * aerodynamics * 0.2f; } } public class Part : MonoBehaviour { public string partName; public float horsepowerBonus; public float weightBonus; public float tractionBonus; public float aeroBonus; } public class Car : MonoBehaviour { public CarStats stats = new CarStats(); public List installedParts = new List(); private Rigidbody rb; private Vector3 startPosition; private Quaternion startRotation; private bool aiEnabled = false; void Start() { rb = GetComponent(); if (rb == null) { Debug.LogError("Rigidbody component missing on car: " + gameObject.name); } startPosition = transform.position; startRotation = transform.rotation; } public void ResetCar() { transform.position = startPosition; transform.rotation = startRotation; rb.velocity = Vector3.zero; rb.angularVelocity = Vector3.zero; } public void EnableAI(bool enable) { aiEnabled = enable; } public void InstallPart(Part part) { if (part == null) { Debug.LogWarning("Attempted to install null part"); return; } installedParts.Add(part); stats.horsepower += part.horsepowerBonus; stats.weight += part.weightBonus; stats.traction += part.tractionBonus; stats.aerodynamics += part.aeroBonus; Debug.Log("Installed: " + part.partName); } void FixedUpdate() { if (rb == null) return; float accel = stats.GetAcceleration(); float topSpeed = stats.GetTopSpeed(); // Proper speed clamping if (rb.velocity.magnitude < topSpeed) { Vector3 forceDirection = aiEnabled ? transform.forward : transform.forward; rb.AddForce(forceDirection * accel, ForceMode.Acceleration); } } }
